int d[16];

void printlist() {
    int i;
    for(i = 0; i < 16; i = i + 1) {
        printf(" ", d[i]);
    }
}

void quicksort(int i, int j) {
    int x, y, z;
    int tmp;

    x = i;
    y = j;
    z = d[(i + j) / 2];

    do {

        if(z < d[y]) {
            do y = y - 1; while(z < d[y])
        }
        if(z > d[x]) {
            do x = x + 1; while(z > d[x])
        }

        tmp = d[x];
        d[x] = d[y];
        d[y] = tmp;
        x = x + 1;
        y = y - 1;

    } while(x <= y)

    if (x<j) quicksort(x,j);
    if (i<y) quicksort(i,y);

}

void main() {
    int i;
    for(i = 0; i < 16; i = i + 1) {
        d[i] = 16 - i;
    }
    quicksort(0, 15);
    printlist();
}
