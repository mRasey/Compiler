int d[10];

void quick_sort(int i, int j) {
    int y, z;
    int tmp;
    x = i;
    y = j;
    z = d[(i + j) / 2];
    do {
        for(y = y; z < d[y]; y = y - 1);
        for(x = x; z > d[x]; x = x + 1);
        if(x <= y) {
            tmp = d[x];
            d[x] = d[y];
            d[y] = tmp;
            x = x + 1;
            y = y - 1;
        }
    } while(x < y)

    if(x < j) quick_sort(x, j);
    if(i < y) quick_sort(i, y);
}

void main() {
    int i;
    for(i = 0; i < 10; i = i + 1) {
        d[i] = 10 - i;
    }
    quick_sort(0, 10);
    for(i = 0; i < 10; i = i + 1) {
        printf(d[i]);
    }
}
